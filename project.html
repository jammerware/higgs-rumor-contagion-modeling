def describe_graph(name, graph):
    print(f'Graph: {name}')
    print(f' - nodes: {len(graph.nodes)}')
    print(f' - edges: {len(graph.edges)}')
    print(f' - avg. clustering coefficient: {len(graph.nodes)}')
    print(f' - assortativity coefficient: {nx.algorithms.assortativity.degree_assortativity_coefficient(graph)}')
    
def initialize_graph(graph, infected_nodes):
    if not isinstance(infected_nodes, list):
        infected_nodes = [infected_nodes]
        
    attrs = { node: { 'state': 'i' } for node in infected_nodes}
    nx.set_node_attributes(graph, 's', name='state')
    nx.set_node_attributes(graph, attrs)
    return graph

def copy_topology(graph):
    new = graph.__class__()
    new.add_nodes_from(graph)
    new.add_edges_from(graph.edges)
    
    return new

def get_nodes_with_state(graph, state):
    return [node for node, attrs in graph.nodes(data=True) if attrs['state'] == state]

def set_nodes_state(graph, nodes, state):
    if not isinstance(nodes, list):
        nodes = [nodes]
        
    for node in nodes:
        nx.set_node_attributes(graph, { node: { 'state': state } })

def infect_simple(graph, beta, infected_nodes, susceptible_nodes):
    nodes_to_infect = []
    
    for susceptible_node in susceptible_nodes:
        roll = random.uniform(0, 1)
        if roll < beta:
            nodes_to_infect.append(susceptible_node)
        
    set_nodes_state(graph, nodes_to_infect, 'i')
        
def infect_complex(graph, infected_nodes, susceptible_nodes):
    nodes_to_infect = []
    
    for susceptible_node in susceptible_nodes:
        neighbors = list(graph.neighbors(susceptible_node))
        infected_neighbors = [neighbor for neighbor in neighbors if neighbor in infected_nodes]
        beta = len(infected_neighbors) / len(neighbors)
        
        if random.uniform(0, 1) < beta:
            nodes_to_infect.append(susceptible_node)
    
    set_nodes_state(graph, nodes_to_infect, 'i')
        

def run_time_step(graph, beta, t, contagion_type='simple'):
    # gather nodes before we mutate so we don't have weird cascading effects
    susceptible_nodes = get_nodes_with_state(graph, 's')
    infected_nodes = get_nodes_with_state(graph, 'i')
    
    # infect
    if contagion_type == 'simple':
        infect_simple(graph, beta, infected_nodes, susceptible_nodes)
    else:
        infect_complex(graph, infected_nodes, susceptible_nodes)
    
    # return snapshot
    total_nodes = len(graph.nodes)
    return {
        't': t,
        'susceptible_pct': len(get_nodes_with_state(graph, 's')) / total_nodes,
        'infected_pct': len(get_nodes_with_state(graph, 'i')) / total_nodes
    }

def run_sim(graph, infected_nodes, beta, max_t=100, contagion_type='simple'):
    df = pd.DataFrame()
    sim_graph = copy_topology(graph)
    sim_graph = initialize_graph(sim_graph, infected_nodes)

    for t in range(max_t):
        snapshot = run_time_step(sim_graph, beta, t)
        df = df.append(snapshot, ignore_index=True)
    
    df.set_index('t', inplace=True)
    return df
